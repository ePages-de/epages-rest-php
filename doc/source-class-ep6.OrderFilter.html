200:     /**
201:      * This function returns the orders by using the order filter.
202:      *
203:      * @author David Pauli <contact@david-pauli.de>
204:      * @since 0.1.3
205:      * @return Order[] Returns an array of orders.
206:      */
207:     public function getOrders() {
208: 
209:         $this->errorReset();
210: 
211:         $parameter = $this->getParameter();
212: 
213:         // if request method is blocked
214:         if (!RESTClient::setRequestMethod(HTTPRequestMethod::GET)) {
215: 
216:             $this->errorSet("RESTC-9");
217:             return;
218:         }
219: 
220:         $content = RESTClient::send(self::RESTPATH . "?" . $parameter);
221: 
222:         // if respond is empty
223:         if (InputValidator::isEmpty($content)) {
224: 
225:             $this->errorSet("OF-1");
226:             Logger::error("ep6\OrderFilter\nREST respomd for getting orders is empty.");
227:             return;
228:         }
229: 
230:         // if there is no results, page AND resultsPerPage element
231:         if (InputValidator::isEmptyArrayKey($content, "results") ||
232:             InputValidator::isEmptyArrayKey($content, "page") ||
233:             InputValidator::isEmptyArrayKey($content, "resultsPerPage")) {
234: 
235:             $this->errorSet("OF-2");
236:             Logger::error("ep6\OrderFilter\nRespond for " . self::RESTPATH . " can not be interpreted.");
237:             return;
238:         }
239: 
240:         $this->results = $content['results'];
241: 
242:         $orders = array();
243: 
244:         // is there any order found: load the products.
245:         if (!InputValidator::isEmptyArrayKey($content, "items") && (sizeof($content['items']) != 0)) {
246: 
247:             foreach ($content['items'] as $item) {
248: 
249:                 $order = new Order($item);
250: 
251:                 // go to every filter
252:                 foreach ($this->filters as $filter) {
253: 
254:                     if (!InputValidator::isEmptyArrayKey($item, $filter->getAttribute()) || $filter->getOperator() == FilterOperation::UNDEF) {
255: 
256:                         if (!InputValidator::isArray($item[$filter->getAttribute()])) {
257: 
258:                             if (!$filter->isElementInFilter($item)) {
259:                                 continue 2;
260:                             }
261:                         }
262:                     }
263:                     else {
264:                         continue 2;
265:                     }
266:                 }
267: 
268:                 array_push($orders, $order);
269:             }
270:         }
271: 
272:         return $orders;
273:     }
